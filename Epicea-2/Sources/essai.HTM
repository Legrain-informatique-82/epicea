procedure TMain.FormCreate(Sender: TObject);
begin

   dbNameDonnees.Close;
   dbNameDonnees.Params[0]:= 'PATH=' + E.RepertoireDonnees;
   dbNameDonnees.Open;

 {Parametre de l'entreprise}
   // A remplir dynamiquement
   with E do begin
      {Monnaie:= 'Fr';}
      ExerciceSaisie:= '98';
      ExerciceBase:= '98';

      Devise:= DeviseFranc;
      ArrondiCredit:= '758';
      ArrondiDebit:= '658';

      DatExoDebut:= StrToDate('01/01/98');
      DatExoFin:= StrToDate('31/12/98');
   end;
end;

procedure TMain.FormShow(Sender: TObject);

      MenuP:=CopyMenu(MenuMain,MenuP,'MenuP');
      Main.Menu:=MenuP;
 end
   else begin
      MenuP := TMainMenu.create(Main);
      ReadComponentResFile('MOrg',MenuP );
